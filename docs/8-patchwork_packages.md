```python
def package(c, *packages):
    apt = 'DEBIAN_FRONTEND=noninteractive apt-get install -y {}'
    yum = 'yum install -y %s'
    manager = apt if distro_family(c) == 'debian' else yum
    for package in packages:
        c.sudo(manager.format(package))
```
